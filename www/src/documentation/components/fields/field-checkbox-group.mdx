Groups of checkboxes can be displayed with `FieldCheckboxGroup`.

There are two "inline" variations:

- The `inline` prop will put the group label and all the inputs in one line (wrapping the inputs as necessary)
- The `inputsInline` prop will stack the group label on top but keep all the inputs inline

```jsx
;() => {
  const options = [
    {
      label: 'Cheddar',
      value: 'cheddar',
    },
    {
      label: 'Gouda',
      value: 'gouda',
    },
  ]

  const optionsDisabled = [
    {
      label: 'Cheddar',
      value: 'cheddar',
    },
    {
      disabled: true,
      label: 'Gouda',
      value: 'gouda',
    },
  ]

  return (
    <Grid>
      <FieldCheckboxGroup
        defaultValue={['cheddar']}
        description="defaultValue checked"
        label="Cheeses default layout"
        name="group1"
        options={options}
      />
      <FieldCheckboxGroup
        defaultValue={['cheddar']}
        description="defaultValue checked"
        inline
        label="Cheeses inline layout"
        name="group1"
        options={options}
      />
      <FieldCheckboxGroup
        description="Validation message and required"
        label="Cheeses default layout"
        name="group1"
        options={options}
        required
        validationMessage={{
          message: 'Select at least 1 cheese',
          type: 'error',
        }}
      />
      <FieldCheckboxGroup
        description="Validation message and required"
        inputsInline
        label="Cheeses inline inputs layout"
        name="group1"
        options={options}
        required
        validationMessage={{
          message: 'Select at least 1 cheese',
          type: 'error',
        }}
      />
      <FieldCheckboxGroup
        description="everything is disabled"
        disabled
        label="Cheeses default layout"
        name="group1"
        options={options}
      />
      <FieldCheckboxGroup
        description="everything is disabled"
        disabled
        inline
        name="group1"
        label="Cheeses inline layout"
        options={options}
      />
      <FieldCheckboxGroup
        defaultValue={['gouda']}
        description="only one checkbox is disabled and defaultValue"
        label="Cheeses default layout"
        name="group1"
        options={optionsDisabled}
      />
      <FieldCheckboxGroup
        defaultValue={['gouda']}
        description="only one checkbox is disabled and defaultValue"
        inline
        name="group1"
        label="Cheeses inline layout"
        options={optionsDisabled}
      />
    </Grid>
  )
}
```
