RadioGroup

STABLE

Groups of radio can be displayed with RadioGroup. Disabled can be used on each radio specifically or on the group.

;() => {
  const options = [
    {
      label: 'Cheddar',
      value: 'cheddar',
    },
    {
      label: 'Gouda',
      value: 'gouda',
    },
  ]


  const optionsDisabled = [
    {
      label: 'Cheddar',
      value: 'cheddar',
    },
    {
      disabled: true,
      label: 'Gouda',
      value: 'gouda',
    },
  ]


  return (
    <Grid m="u10">
      <RadioGroup defaultValue={'cheddar'} name="group1" options={options} />
      <RadioGroup inline name="group1" options={options} />
      <RadioGroup name="group1" options={optionsDisabled} />
      <RadioGroup inline name="group1" options={optionsDisabled} />
      <RadioGroup disabled name="group1" options={options} />
      <RadioGroup disabled inline name="group1" options={options} />
    </Grid>
  )
}

Controlled

Control externally via value and onChange props.

Current selection: cheddar

;() => {
  const options = [
    {
      label: 'Cheddar',
      value: 'cheddar',
    },
    {
      label: 'Gouda',
      value: 'gouda',
    },
  ]
  const [value, setValue] = useState('cheddar')
  return (
    <Space align="start">
      <RadioGroup
        name="cheeses"
        value={value}
        onChange={setValue}
        options={options}
      />
      <Paragraph>Current selection: {value}</Paragraph>
    </Space>
  )
}
