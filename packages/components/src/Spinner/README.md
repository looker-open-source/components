# Spinner

```tsx
function Basic() {
  return <Spinner />;
}
```

## Set color

```tsx
function Color() {
  return <Spinner color="salmon" />;
}
```

## Set size

```tsx
function Size() {
  return <Spinner size={80} />;
}
```

## Set speed

```tsx
function Speed() {
  return <Spinner speed={2000} />;
}
```

## Set markers

```tsx
function Markers() {
  return <Spinner markers={20} markerRadius={50} />;
}
```
